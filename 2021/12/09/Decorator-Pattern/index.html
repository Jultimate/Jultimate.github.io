Buff

public abstract class Buff
{
    protected Legend legend;

    protected abstract void doBuff();
}

public class RedBuff extends Buff
{
    public RedBuff(Legend legend)
    {
        this.legend = legend;
        doBuff();
    }

    @Override
    protected void doBuff()
    {
        legend.dot += 10;
        System.out.println("Legend have red buff.");
    }
}

public class BlueBuff extends Buff
{
    public BlueBuff(Legend legend)
    {
        this.legend = legend;
        doBuff();
    }

    @Override
    protected void doBuff()
    {
        legend.coldDown *= 0.75; 
        System.out.println("Legend have blue buff.");
    }
}
